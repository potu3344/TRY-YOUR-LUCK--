  <script>
    let namesList = [];

    let input = document.getElementById("nameInput");

    input.addEventListener("keydown", function(e) {
      if (e.key === "Enter" || e.key === ",") {
        e.preventDefault();

        let value = input.value.trim();
        if (value === "") return;

        addName(value);
        input.value = "";
      }
    });

    function addName(name) {
      let lower = name.toLowerCase();

      if (namesList.map(n => n.toLowerCase()).includes(lower)) {
        alert("⚠️ Duplicate name!");
        return;
      }

      namesList.push(name);
      renderTags();
    }

    function renderTags() {
      let tagsDiv = document.getElementById("tags");
      tagsDiv.innerHTML = "";

      namesList.forEach((n, index) => {
        let tag = document.createElement("div");
        tag.className = "tag";
        tag.innerHTML = `${index + 1}. ${n} <span onclick="removeTag(${index})">×</span>`;
        tagsDiv.appendChild(tag);
      });
    }

    function removeTag(index) {
      namesList.splice(index, 1);
      renderTags();
    }

    function setAmount(val) {
      document.getElementById("amount").value = val;
    }

    function startLottery() {
      let names = namesList;
      let amount = parseFloat(document.getElementById("amount").value);
      let winnerCount = parseInt(document.getElementById("winnerCount").value);

      if (names.length < winnerCount) {
        alert("Players kom ache!");
        return;
      }

      if (!amount || amount <= 0) {
        alert("Valid amount dao!");
        return;
      }

      let loading = document.getElementById("loading");
      let result = document.getElementById("result");

      result.innerHTML = "";
      loading.style.display = "block";

      let i = 0;

      function roll() {
        let randomName = names[Math.floor(Math.random() * names.length)];
        loading.innerText = "🎡 " + randomName;

        i++;

        if (i < 40) {
          setTimeout(roll, 40 + i * 10);
        } else {
          finishLottery(names, amount, winnerCount);
        }
      }

      roll();
    }

    function finishLottery(names, amount, winnerCount) {
      let loading = document.getElementById("loading");
      let result = document.getElementById("result");

      let shuffled = [...names].sort(() => Math.random() - 0.5);

      let winners = shuffled.slice(0, winnerCount);
      let losers = shuffled.slice(winnerCount);

      let split = (amount / winnerCount).toFixed(2);

      loading.style.display = "none";

      result.innerHTML = `💰 Total: ₹${amount}<br><br>🏆 Winners:<br>`;

      winners.forEach((w, i) => {
        setTimeout(() => {
          result.innerHTML += `✨ <span class="winner-name">${w}</span> → <span class="winner-amount">₹${split}</span><br>`;
        }, i * 500);
      });

      setTimeout(() => {
        result.innerHTML += `<br>💀 Losers:<br>`;

        losers.forEach((l, i) => {
          setTimeout(() => {
            result.innerHTML += `🎰 <span class="loser-name">${l}</span><br>`;
          }, i * 300);
        });
      }, winners.length * 500 + 500);
    }
